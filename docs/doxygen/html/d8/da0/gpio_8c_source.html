62/***************************************************************************************************
63 * @details This function configures the mode of a GPIO pin on the STM32F10RB microcontroller.
64 * The pin number should be within the range 0-15.
65 * The GPIO port is checked for initialization, and if not initialized, it is enabled.
66 * The pin mode is then set according to the provided mode.
67 * The mode parameter should be one of the values from the GpioMode enumeration.
68 * Upon successful configuration, the function returns 0. Otherwise, 1 is returned and
69 * variables errnum and errcode are set with the error code.
70 *
71 * @todo Add option to configure port speed
72 */
73int gpioPinSetup(GPIO_Type *port, uint8_t pin, GpioMode mode) {
74 if (pin > 15) {
75 errnum = 1; // GPIO error
76 errcode = 2; // Wrong pin number
77 return 1;
78 }
79
80 // Check if port is initializated, if not, initialize it
81 if (port == GPIOA) {
82 if (!(RCC->AHB1ENR & (1 << 0))) {
83 RCC->AHB1ENR |= (1 << 0);
84 delay();
85 }
86 } else if (port == GPIOB) {
87 if (!(RCC->AHB1ENR & (1 << 1))) {
88 RCC->AHB1ENR |= (1 << 1);
89 delay();
90 }
91 }
92 else if (port == GPIOC) {
93 if (!(RCC->AHB1ENR & (1 << 2))) {
94 RCC->AHB1ENR |= (1 << 2);
95 delay();
96 }
97 }
98 else if (port == GPIOH) {
99 if (!(RCC->AHB1ENR & (1 << 7))) {
100 RCC->AHB1ENR |= (1 << 7);
101 delay();
102 }
103 }
104 else {
105 errnum = 1; // GPIO error
106 errcode = 1; // Wrong GPIO port
107 return 1;
108 }
109
110 port->MODER &= ~(3 << pin * 2); // Reset mode before overwritting with new one
111 port->MODER |= (mode << pin * 2);
112
113 return 0;
114}

142/***************************************************************************************************
143 * @details This function reads the value of a GPIO pin on the STM32F10RB microcontroller.
144 * The pin number should be within the range 0-15.
145 * The function checks if the provided GPIO port is correct and initialized,
146 * as well as the mode of the pin.
147 * Upon successful read, the function stores the read value in the memory location
148 * pointed to by the 'read' parameter.
149 * The read value will be either 0 or 1, indicating the logic level of the pin.
150 * If an error occurs, 1 is returned, and variables errnum and errcode are set with
151 * the error code.
152 *
153 * @todo Check if the GPIO port is correct and if it is initialized and in the appropriate
154 * mode.
155 */
156int gpioPinRead(GPIO_Type *port, uint8_t pin, uint8_t *read) {
157 if (pin > 15) {
158 errnum = 1; // GPIO error
159 errcode = 2; // Wrong pin number
160 return 1;
161 }
162
163 uint32_t pin_value = (port->IDR >> pin) & 0x01;
164 *read = (uint8_t) pin_value;
165 return 0;
166}

169/***************************************************************************************************
170 * @details This function writes a value (0 or 1) to a GPIO pin on the STM32F10RB
171 * microcontroller.
172 * The pin number should be within the range 0-15.
173 * The function checks if the provided GPIO port is correct and initialized,
174 * as well as the mode of the pin.
175 * The previous value of the pin is stored in the memory location pointed to by the
176 * 'old_value' parameter.
177 * If the 'value' parameter is 0, the pin is cleared (set to logic low).
178 * If the 'value' parameter is 1, the pin is set (set to logic high).
179 * If an error occurs, 1 is returned, and variables errnum and errcode are set with
180 * the error code.
181 *
182 * @todo Check if the GPIO port is correct and if it is initialized and in the appropriate
183 * mode.
184 */
185int gpioPinWrite(GPIO_Type *port, uint8_t pin, uint8_t value, uint8_t *old_value) {
186 if (pin > 15) {
187 errnum = 1; // GPIO error
188 errcode = 2; // Wrong pin number
189 return 1;
190 }
191
192 uint32_t pin_value = (port->IDR >> pin) & 0x01;
193 *old_value = (uint8_t) pin_value;
194
195 if (value == 0) {
196 port->ODR &= ~(1 << pin);
197 } else if (value == 1) {
198 port->ODR |= (1 << pin);
199 } else {
200 errnum = 1; // GPIO error
201 errcode = 5; // Tried to write wrong value
202 return 1;
203 }
204
205 return 0;
206}

209/***************************************************************************************************
210 * @details This function toggles the value of a GPIO pin on the STM32F10RB microcontroller.
211 * The pin number should be within the range 0-15.
212 * The function checks if the provided GPIO port is correct and initialized,
213 * as well as the mode of the pin.
214 * The previous value of the pin is stored in the memory location pointed to by the
215 * 'old_value' parameter.
216 * If the pin is currently low, it is toggled to high. If the pin is currently high,
217 * it is toggled to low.
218 * If an error occurs, 1 is returned, and variables errnum and errcode are set with
219 * the error code.
220 *
221 * @todo Check if the GPIO port is correct and if it is initialized and in the appropriate mode.
222 */
223int gpioPinToggle(GPIO_Type *port, uint8_t pin, uint8_t *old_value) {
224 if (pin > 15) {
225 errnum = 1; // GPIO error
226 errcode = 2; // Wrong pin number
227 return 1;
228 }
229
230 uint32_t pin_value = (port->IDR >> pin) & 0x01;
231 *old_value = (uint8_t) pin_value;
232
233 port->ODR ^= (1 << pin);
234 return 0;
235}

238/***************************************************************************************************
239 * @details This function sets up an interrupt for a GPIO pin on the STM32F10RB microcontroller.
240 * The pin number should be within the range 0-15.
241 * The function checks if the provided GPIO port is correct and initialized.
242 * It also validates the rising_edge and priority parameters.
243 * The interrupt trigger type is set based on the rising_edge parameter,
244 * with 1 indicating a rising edge trigger and 0 indicating a falling edge trigger.
245 * The priority parameter sets the interrupt priority level, which should be between
246 * 7 and 15 (inclusive).
247 * The handler parameter is a pointer to the interrupt handler function that will be
248 * executed when the interrupt occurs.
249 * Upon successful setup, the function returns 0.
250 * If an error occurs, 1 is returned, and variables errnum and errcode are set with
251 * the appropriate error codes.
252 *
253 * @todo Check if the GPIO port is correct and if it is initialized.
254 * @todo Clarify the allowed range for the priority parameter and which value is higher prio
255 */
256int gpioInterruptSet(GPIO_Type *port, uint8_t pin, uint8_t rising_edge, uint8_t priority, void (*handler)(void)) {
257 if (pin > 15) {
258 errnum = 1; // GPIO error
259 errcode = 2; // Wrong pin number
260 return 1;
261 }
262 if (rising_edge > 1) {
263 errnum = 1; // GPIO error
264 errcode = 6; // Wrong value for trigger selection
265 return 1; // Can only be 0 or 1
266 }
267 if (priority < 7) {
268 errnum = 1; // GPIO error
269 errcode = 7: // Wrong interrupt priority
270 return 1;
271 }
272
273 uint8_t exti_source_input;
274
275 if (port == GPIOA)
276 exti_source_input = 0;
277 else if (port == GPIOB)
278 exti_source_input = 1;
279 else if (port == GPIOC)
280 exti_source_input = 2;
281 else if (port == GPIOH)
282 exti_source_input = 7;
283 else {
284 errnum = 1; // GPIO error
285 errcode = 1; // Wrong GPIO port
286 return 1;
287 }
288
289 RCC->APB2ENR |= (1 << 14); // Enable system configuration controller clock
290 SYSCFG->EXTICR[pin / 4] &= ~(0xF << (pin % 4) * 4); // Clear register
291 SYSCFG->EXTICR[pin / 4] |= (exti_source_input << (pin % 4) * 4); // Select source input for the EXTIx
292
293 EXTI->IMR |= (1 << pin); // Set pin in EXTI line as interrupt
294 if (rising_edge == 1) {
295 EXTI->RTSR |= (1 << pin); // Enable rising edge trigger
296 EXTI->FTSR &= ~(1 << pin); // Disable falling edge trigger
297 } else {
298 EXTI->RTSR &= ~(1 << pin); // Disable rising edge trigger
299 EXTI->FTSR |= (1 << pin); // Enable falling edge trigger
300 }
301
302 // Set priority and set-enable interrupt
303 if (pin <= 4) {
304 NVIC->IPR[6 + pin] |= (priority << 4);
305 NVIC->ISER[0] |= (1 << (6 + pin));
306 } else if (pin <= 9) {
307 NVIC->IPR[23] |= (priority << 4);
308 NVIC->ISER[0] |= (1 << 23);
309 } else {
310 NVIC->IPR[40] |= (priority << 4);
311 NVIC->ISER[1] |= (1 << 8);
312 }
313
314 gpio_isr_functions[pin] = handler;
315
316 return 0;
317}

320/***************************************************************************************************
321 * @brief Interrupt Service Routine for EXTI0.
322 *
323 * @details This ISR is triggered when an interrupt occurs on EXTI line 0.
324 * It calls the associated GPIO interrupt handler function stored in
325 * gpio_isr_functions[0].
326 * After handling the interrupt, it clears the interrupt flag for EXTI line 0.
327 */
328static void EXTI0_ISR (void) {
329 if (EXTI->PR & (1 << 0)) {
330 gpio_isr_functions[0]();
331 EXTI->PR |= (1 << 0); // Clear flag
332 }
333}
