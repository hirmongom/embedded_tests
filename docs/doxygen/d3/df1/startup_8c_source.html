269 void Reset_ISR(void) {
270  // Copy .data section to SRAM
271  uint32_t size = (uint32_t)&_edata - (uint32_t)&_sdata;
272  uint8_t *pDestination = (uint8_t*)&_sdata; // SRAM
273  uint8_t *pSource = (uint8_t*)&_etext; // FLASH
274 
275  asm volatile(
276  "mov r0, %[pDestination]\n" // Load the start address of the destination into r0
277  "mov r1, %[size]\n" // Load the size of the copy into r1
278  "mov r2, #0\n" // Initialize r2 with zero
279 
280  "copy_loop:\n"
281  "cmp r1, #0\n" // Compare the size with zero
282  "beq copy_end\n" // Branch to copy_end if size is zero
283  "ldrb r3, [%[pSource]]\n" // Load a byte from the source address
284  "strb r3, [r0], #1\n" // Store the byte to the destination address and increment the destination pointer
285  "subs r1, r1, #1\n" // Subtract 1 from size
286  "b copy_loop\n" // Branch back to copy_loop
287 
288  "copy_end:\n"
289  :
290  : [pDestination] "r" (pDestination), [size] "r" (size), [pSource] "r" (pSource)
291  : "r0", "r1", "r2", "r3"
292  );
293 
294  // Initialize .bss section to zero
295  size = (uint32_t)&_ebss - (uint32_t)&_sbss;
296  pDestination = (uint8_t*)&_sbss;
297 
298  asm volatile(
299  "mov r0, %[dest]\n" // Load the start address of .bss section into r0
300  "mov r1, %[size]\n" // Load the size of .bss section into r1
301  "mov r2, #0\n" // Initialize r2 with zero
302 
303  "loop:\n"
304  "cmp r1, #0\n" // Compare the size with zero
305  "beq end\n" // Branch to end if size is zero
306  "strb r2, [r0], #1\n" // Store byte zero at the current address and increment the address
307  "subs r1, r1, #1\n" // Subtract 1 from size
308  "b loop\n" // Branch back to the loop
309 
310  "end:\n"
311  :
312  : [dest] "r" (pDestination), [size] "r" (size)
313  : "r0", "r1", "r2"
314  );
315
