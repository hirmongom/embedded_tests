Default Interrupt Service Routine (ISR)

This ISR is a placeholder for all other interrupts that do not have a dedicated ISR defined. It enters an infinite loop, effectively halting the system when an unexpected interrupt occurs.

void Reset_ISR ( void  )

Reset Interrupt Service Routine (ISR)

This ISR is invoked when a reset occurs. It is responsible for initializing the .data section by copying the data from FLASH to SRAM, initializing the .bss section by setting it to zero, and then calling the user-defined main function.

291 {
292 // Copy .data section to SRAM
293 uint32_t size = (uint32_t)&_edata - (uint32_t)&_sdata;
294 uint8_t *pDestination = (uint8_t*)&_sdata; // SRAM
295 uint8_t *pSource = (uint8_t*)&_etext; // FLASH
296
297 asm volatile(
298 "mov r0, %[pDestination]\n" // Load the start address of the destination into r0
299 "mov r1, %[size]\n" // Load the size of the copy into r1
300 "mov r2, #0\n" // Initialize r2 with zero
301
302 "copy_loop:\n"
303 "cmp r1, #0\n" // Compare the size with zero
304 "beq copy_end\n" // Branch to copy_end if size is zero
305 "ldrb r3, [%[pSource]]\n" // Load a byte from the source address
306 "strb r3, [r0], #1\n" // Store the byte to the destination address and increment the destination pointer
307 "subs r1, r1, #1\n" // Subtract 1 from size
308 "b copy_loop\n" // Branch back to copy_loop
309
310 "copy_end:\n"
311 :
312 : [pDestination] "r" (pDestination), [size] "r" (size), [pSource] "r" (pSource)
313 : "r0", "r1", "r2", "r3"
314 );
315
316 // Initialize .bss section to zero
317 size = (uint32_t)&_ebss - (uint32_t)&_sbss;
318 pDestination = (uint8_t*)&_sbss;
319
320 asm volatile(
321 "mov r0, %[dest]\n" // Load the start address of .bss section into r0
322 "mov r1, %[size]\n" // Load the size of .bss section into r1
323 "mov r2, #0\n" // Initialize r2 with zero
324
325 "loop:\n"
326 "cmp r1, #0\n" // Compare the size with zero
327 "beq end\n" // Branch to end if size is zero
328 "strb r2, [r0], #1\n" // Store byte zero at the current address and increment the address
329 "subs r1, r1, #1\n" // Subtract 1 from size
330 "b loop\n" // Branch back to the loop
331
332 "end:\n"
333 :
334 : [dest] "r" (pDestination), [size] "r" (size)
335 : "r0", "r1", "r2"
336 );
337
338 // Call program entry point
339 main();
340}
